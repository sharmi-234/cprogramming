#include <stdio.h>
void swap(int *a,int *b){
    int temp;
    temp=*a;
    *a=*b;
    *b=temp;
    
}
int main(){
    int x,y;
    printf("Enter two integers:\n");
    scanf("%d %d",&x,&y);
     
    printf("Before swapping:\n");
    printf("x=%d,y=%d\n",x,y);
    swap(&x,&y);
    printf("\nAfter swapping:\n");
    printf("x=%d,y=%d\n",x,y);
    return 0;
    
}
